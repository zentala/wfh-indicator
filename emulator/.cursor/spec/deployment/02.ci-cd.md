### GitHub Actions

#### Main Workflow
```yaml
# .github/workflows/main.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      - name: Install pnpm
        run: npm install -g pnpm
      - name: Install dependencies
        run: pnpm install
      - name: Run linting
        run: pnpm lint
      - name: Run tests
        run: pnpm test
      - name: Build application
        run: pnpm build
```

#### E2E Tests Workflow
```yaml
# .github/workflows/e2e.yml
name: E2E Tests

on:
  push:
    branches: [main]

jobs:
  e2e:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Install dependencies
        run: |
          cd emulator && pnpm install
          cd ../tray-app && pnpm install
      - name: Start mock device
        run: |
          cd emulator
          pnpm start:mock &
          sleep 5
      - name: Run E2E tests
        run: |
          cd tray-app
          pnpm test:e2e
```
